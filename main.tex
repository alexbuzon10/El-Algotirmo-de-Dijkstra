\section{Introducción}

\begin{frame}{Introducción}
    \begin{block}{Pregunta inicial}
        Si existen varios caminos para llegar de un punto a otro, ¿cómo podemos determinar cuál es el más eficiente?
    \end{block}

    \begin{itemize}
        \item En muchos problemas no basta con llegar al destino.
        \item También interesa minimizar la distancia, el tiempo, el coste o los recursos.
        \item Este desafío se conoce como el \textbf{problema del camino mínimo}.
    \end{itemize}

    \vspace{0.4cm}
    El algoritmo de Dijkstra permite abordar este problema en grafos ponderados con pesos no negativos.
\end{frame}

\begin{frame}{Motivación del problema}
    El problema del camino mínimo aparece en multitud de situaciones reales:

    \begin{itemize}
        \item \textbf{Sistemas GPS:} para calcular rutas eficientes.
        \item \textbf{Redes informáticas:} para enviar datos por los canales adecuados.
        \item \textbf{Logística y transporte:} para reducir costes y tiempos de entrega.
        \item \textbf{Planificación urbana:} para trazar recorridos en mapas o redes.
    \end{itemize}

    \vspace{0.4cm}
    Por ello, su estudio es fundamental tanto en la Matemática Discreta como en la Informática.
\end{frame}

\section{Descripción del algoritmo}

\begin{frame}{Descripción del algoritmo}
    El algoritmo de Dijkstra es un método basado en la teoría de grafos. Un grafo representa las relaciones entre elementos mediante:
    
    \begin{itemize}
        \item \textbf{Vértices (o nodos):} los puntos de conexión.
        \item \textbf{Aristas:} las uniones entre nodos.
    \end{itemize}

    \begin{block}{Grafo ponderado}
        Es aquel donde a cada arista se le asigna un valor numérico llamado \textbf{"peso"} (o coste), que indica la importancia o el coste de transitarla.
    \end{block}
\end{frame}

\begin{frame}{Historia y Estrategia}
    La solución más eficiente a este problema la aportó \textbf{Edsger Dijkstra en 1956}. 
    
    \vspace{0.3cm}
    Su algoritmo opera bajo la condición de que todos los pesos de la red sean positivos y utiliza una estrategia de \textbf{"pasos seguros"}:

    \begin{enumerate}
        \item \textbf{Inicialización}: Distancia origen = 0, resto = $\infty$.
        \item \textbf{Elección}: Selecciona el nodo no visitado con menor coste acumulado.
        \item \textbf{Relajación}: Evalúa si pasar por el nodo actual acorta la distancia a sus vecinos.
        \item \textbf{Cierre}: Marca el nodo como definitivo.
    \end{enumerate}
\end{frame}

\section{Fundamento matemático}

\begin{frame}{Fundamento matemático}
    \begin{block}{Definición (Grafo ponderado)}
        Un grafo ponderado es un par $(G, w)$, donde $G=(V,E)$ es un grafo y $w:E\rightarrow\mathbb{R}^{+}$ es una función de peso.
    \end{block}
    
    \begin{block}{Peso de un camino $C$}
        Para un camino $C: v_{0}v_{1}\dots v_{k}$, el peso es:
        \[ w(C)=\sum_{i=1}^{k}w(v_{i-1},v_{i}) \]
    \end{block}
    
    \begin{block}{Distancia entre vértices}
        \[ d_{G}(u,v)=\min\{w(C) \mid C \text{ es un } u\text{-}v \text{ camino}\} \]
        Si no existe camino, $d_{G}(u,v)=+\infty$.
    \end{block}
\end{frame}

\begin{frame}{Principio de optimalidad y Relajación}
    \textbf{Principio de optimalidad de Bellman}\\
    Cualquier subcamino de un camino mínimo es también un camino mínimo.

    \vspace{0.4cm}
    \textbf{Operación de relajación}\\
    Para cada arista $(u, v) \in E$:
    \begin{exampleblock}{Actualización de distancia}
        \[ \text{si } d[u] + w(u,v) < d[v] \implies d[v] \gets d[u] + w(u,v) \]
    \end{exampleblock}
\end{frame}

\begin{frame}{Complejidad computacional}
    El algoritmo requiere pesos no negativos ($w:E\rightarrow\mathbb{R}^{+}$).
    
    \begin{table}
        \centering
        \renewcommand{\arraystretch}{1.2}
        \begin{tabular}{lc}
            \hline
            \textbf{Estructura de datos} & \textbf{Complejidad temporal} \\ \hline
            Lista o Arreglo simple       & $O(|V|^2)$ \\
            Heap binario                 & $O((|V| + |E|) \log |V|)$ \\
            Heap de Fibonacci            & $O(|E| + |V| \log |V|)$ \\ \hline
        \end{tabular}
        \caption{Eficiencia según la implementación}
    \end{table}
\end{frame}

                \item $(A, C) \rightarrow 2$

\begin{frame}{Ejemplo: Desarrollo (I)}
    \textbf{Paso 1: Seleccionamos Nodo A (Origen)}
    \begin{itemize}
        \item Distancias iniciales: $A=0$, resto $\infty$.
        \item Vecinos de $A$: $B$ y $C$.
        \item $B: 0+5=5 < \infty$ (Actualizamos).
        \item $C: 0+2=2 < \infty$ (Actualizamos).
    \end{itemize}
    \centering
    \begin{tikzpicture}[scale=0.8]
        \node[visitado] (A) at (0,0) {A};
        \node[nodo] (B) at (2.5,0) {B};
        \node[nodo] (C) at (0,-2) {C};
        \node[nodo] (D) at (2.5,-2) {D};
        \path[flecha] (A) edge node[above] {5} (B) (A) edge node[left] {2} (C) (B) edge node[right] {1} (D) (C) edge node[below] {2} (B) (C) edge node[below] {7} (D);
    \end{tikzpicture}
\end{frame}

\begin{frame}{Ejemplo: Desarrollo (II)}
    \textbf{Paso 2: Seleccionamos Nodo C (Distancia 2)}
    \begin{itemize}
        \item Elegimos $C$ porque $2 < 5$ ($B$).
        \item Vecinos de $C$: $B$ y $D$.
        \item $B: 2+2=4$. Como $4 < 5$, \textbf{actualizamos B}.
        \item $D: 2+7=9 < \infty$ (Actualizamos).
    \end{itemize}
    \centering
    \begin{tikzpicture}[scale=0.8]
        \node[visitado] (A) at (0,0) {A};
        \node[nodo] (B) at (2.5,0) {B};
        \node[visitado] (C) at (0,-2) {C};
        \node[nodo] (D) at (2.5,-2) {D};
        \path[flecha] (A) edge node[above] {5} (B) (A) edge node[left] {2} (C) (B) edge node[right] {1} (D) (C) edge node[below] {2} (B) (C) edge node[below] {7} (D);
    \end{tikzpicture}
\end{frame}

\begin{frame}{Ejemplo: Desarrollo (III)}
    \textbf{Paso 3: Seleccionamos Nodo B (Distancia 4)}
    \begin{itemize}
        \item Vecino de $B$: $D$.
        \item $D: 4+1=5$. Como $5 < 9$, \textbf{actualizamos D}.
    \end{itemize}
    
    \vspace{0.2cm}
    \textbf{Paso 4: Seleccionamos Nodo D (Distancia 5)}
    \begin{itemize}
        \item No hay más vecinos no visitados. Algoritmo finalizado.
    \end{itemize}
    \centering

\begin{frame}{Resumen de Resultados}
    \begin{table}
        \centering
        \begin{tabular}{|c|c|l|}
            \hline
            \textbf{Nodo} & \textbf{Distancia Mínima} & \textbf{Camino (Predecesores)} \\ \hline
            A & 0 & - \\ \hline
            B & 4 & $A \to C \to B$ \\ \hline
            C & 2 & $A \to C$ \\ \hline
            D & 5 & $A \to C \to B \to D$ \\ \hline
        \end{tabular}
    \end{table}
    \begin{block}{Conclusión del ejemplo}
        La ruta óptima para llegar a $D$ no es la directa, sino pasando por $C$ y $B$, logrando un coste total de 5 frente a opciones más pesadas.
    \end{block}
\end{frame}

\section{Pseudocódigo}

\begin{frame}[fragile]{Pseudocódigo del algoritmo}
\begin{exampleblock}{Función Dijkstra(Grafo, origen)}
\begin{small}
\begin{verbatim}
dist[origen] = 0
Para cada vértice v en Grafo:
    Si v != origen:
        dist[v] = infinito
    Añadir v a Q (conjunto no visitados)

Mientras Q no esté vacío:
    u = vértice en Q con mínima dist[u]
    Eliminar u de Q
    Para cada vecino v de u todavía en Q:
        alt = dist[u] + peso(u, v)
        Si alt < dist[v]:
            dist[v] = alt
            prev[v] = u
Devolver dist, prev
\end{verbatim}
\end{small}
\end{exampleblock}
\end{frame}
