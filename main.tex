    \section{Fundamento matemático}

    \begin{frame}{Fundamento matemático}
        Para comprender el algoritmo de Dijkstra desde una perspectiva rigurosa, es necesario establecer los fundamentos que garantizan su correctitud. Esta sección presenta las definiciones formales y las propiedades estructurales en las que se basa el algoritmo.
        
        \begin{block}{Definición (Grafo ponderado)}
        Un grafo ponderado es un par $(G, w)$, donde $G=(V,E)$ es un grafo y $w:E\rightarrow\mathbb{R}^{+}$ es una función que asigna pesos positivos a las aristas del grafo.
        \end{block}
    \end{frame}

    \begin{frame}{Fundamento matemático: Camino y Distancia}
        \begin{block}{Definición (Camino y su peso)}
        Dado un grafo ponderado $(G, w)$ y un camino $C:v_{0}v_{1}\dots v_{k}$, se define el peso del camino $C$ como la suma de los pesos de las aristas que lo componen:
        \[
        w(C)=\sum_{i=1}^{k}w(v_{i-1},v_{i})
        \]
        \end{block}
        
        \begin{block}{Definición (Distancia entre vértices)}
        Para dos vértices distintos $u, v \in V$, la distancia $d_{G}(u,v)$ es:
        \[
        d_{G}(u,v)=\min\{w(C) \mid C \text{ es un } u\text{-}v \text{ camino}\}
        \]
        Si no hay camino, $d_{G}(u,v)=+\infty$.
        \end{block}
    \end{frame}

    \begin{frame}{Principio de optimalidad y Relajación}
        \textbf{Principio de optimalidad de Bellman}\\
        Si un camino es mínimo, cualquier subcamino contenido en él también es un camino mínimo entre sus propios extremos.

        \begin{proof}
        Si existiera un subcamino más corto, el peso total del camino original podría reducirse, lo que contradice que era el mínimo.
        \end{proof}
        
        \vspace{0.3cm}
        \textbf{Operación de relajación}\\
        El algoritmo mantiene para cada vértice $v$ una estimación de su distancia mínima $d[v]$ desde la fuente $s$ (inicialmente $d[s] = 0$ y $d[v] = +\infty$ para el resto).
        
        Para cada arista $(u, v) \in E$:
        \[
        \text{si } d[u] + w(u,v) < d[v] \implies d[v] \gets d[u] + w(u,v)
        \]
    \end{frame}

    \begin{frame}{Condiciones y Complejidad}
        \textbf{Condición de pesos no negativos}\\
        Requiere que la función de peso sea $w:E\rightarrow\mathbb{R}^{+}$. Pesos negativos invalidarían la estrategia \textit{greedy}, ya que una distancia "finalizada" podría reducirse posteriormente por un arco negativo.
        
        \vspace{0.3cm}
        \textbf{Complejidad computacional}
        \begin{table}[h]
        \centering
        \begin{tabular}{|l|c|}
        \hline
        \textbf{Estructura de datos} & \textbf{Complejidad temporal} \\
        \hline
        Lista o Arreglo simple & $O(|V|^2)$ \\
        Heap binario & $O((|V| + |E|) \log |V|)$ \\
        Heap de Fibonacci & $O(|E| + |V| \log |V|)$ \\
        \hline
        \end{tabular}
        \end{table}
    \end{frame}

    \section{Explicación del pseudocódigo}

    \begin{frame}{Pseudocódigo del algoritmo}
        \texttt{
            \noindent function Dijkstra(Graph, source):\\
            \qquad dist[source] = 0 \\
            \qquad for each vertex v in Graph: \\
            \qquad  \qquad if v != source: \\
            \qquad \qquad \qquad dist[v] = infinity \\ 
            \qquad  \qquad add v to Q (unvisited set) \\ 
            \qquad while Q is not empty: \\
            \qquad \qquad u = vertex in Q with min dist[u] \\ 
            \qquad \qquad remove u from Q \\ 
            \qquad \qquad for each neighbor v of u in Q: \\
            \qquad \qquad \qquad alt = dist[u] + length(u, v) \\ 
            \qquad \qquad \qquad if alt < dist[v]: \\
            \qquad \qquad \qquad \qquad dist[v] = alt \\
            \qquad \qquad \qquad \qquad prev[v] = u \\
            \qquad return dist, prev
        }
    \end{frame}
    
        \begin{frame}{Explicación del pseudocódigo}
        El algoritmo de Dijkstra calcula la ruta más corta desde un nodo inicial hacia todos los demás nodos del grafo. Requiere que todas las distancias sean positivas o cero. Su idea principal es seleccionar siempre el nodo no visitado más cercano al origen y actualizar las distancias de sus nodos vecinos si encuentra un camino más corto pasando por él.
    \end{frame}

    \begin{frame}{Explicación del pseudocódigo: línea por línea I}
        \begin{itemize}
            \item \texttt{dist[source] = 0}: La distancia para ir del punto de inicio a sí mismo es cero.
            
            \item \texttt{for each vertex v in Graph}: Recorre todos los demás puntos o nodos de la red.
            
            \item \texttt{dist[v] = infinity}: Como al principio el algoritmo no sabe cómo llegar al resto de nodos, les asigna temporalmente una distancia infinita.
            
            \item \texttt{add v to Q}: Introduce todos los nodos en un conjunto llamado \texttt{Q}, que funciona como una lista de tareas de nodos no visitados.
        \end{itemize}
    \end{frame}

    \begin{frame}{Explicación del pseudocódigo: línea por línea II}
        \begin{itemize}
            \item \texttt{while Q is not empty}: El proceso se repite en bucle hasta que la lista de nodos no visitados quede vacía.
            
            \item \texttt{u = vertex in Q with min dist[u]}: Selecciona de la lista el nodo que tenga la menor distancia acumulada. En la primera iteración, este será necesariamente el nodo origen.
            
            \item \texttt{remove u from Q}: Elimina ese nodo de la lista para marcarlo oficialmente como visitado.
            
            \item \texttt{for each neighbor v of u in Q}: Revisa únicamente los nodos vecinos de \texttt{u} que todavía no hayan sido visitados.
        \end{itemize}
    \end{frame}

    \begin{frame}{Explicación del pseudocódigo: línea por línea III}
        \begin{itemize}
            \item \texttt{alt = dist[u] + length(u, v)}: Calcula la longitud de una posible nueva ruta hasta \texttt{v}, sumando la distancia ya conocida hasta \texttt{u} más el peso de la arista entre \texttt{u} y \texttt{v}.
            
            \item \texttt{if alt < dist[v]}: Este es el núcleo del algoritmo. Comprueba si la nueva distancia calculada es menor que la mejor distancia conocida hasta ese momento para llegar a \texttt{v}.
            
            \item \texttt{dist[v] = alt} y \texttt{prev[v] = u}: Si la nueva ruta es mejor, se actualiza la distancia mínima registrada y se guarda que el nodo anterior óptimo para llegar a \texttt{v} es \texttt{u}.
        \end{itemize}
    \end{frame}

    \begin{frame}{Resultado final del algoritmo}
        \begin{itemize}
            \item \texttt{return dist, prev}: El algoritmo finaliza devolviendo dos estructuras:
            \begin{itemize}
                \item \texttt{dist}, que contiene la distancia mínima desde el nodo origen hasta cada nodo del grafo.
                \item \texttt{prev}, que permite reconstruir el camino más corto seguido para llegar a cada nodo.
            \end{itemize}
        \end{itemize}
        
        \vspace{0.3cm}
        Gracias a estas dos estructuras, no solo se conoce cuánto cuesta llegar a cada punto, sino también cuál es la ruta exacta que debe seguirse.
    \end{frame}
